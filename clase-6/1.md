```sql
-- 1. USAR MASTER Y ELIMINAR LA BASE DE DATOS SI EXISTE
USE master;
GO

IF DB_ID('RedSocialDB') IS NOT NULL
BEGIN
    ALTER DATABASE RedSocialDB SET SINGLE_USER WITH ROLLBACK IMMEDIATE;
    DROP DATABASE RedSocialDB;
    PRINT 'Base de datos RedSocialDB eliminada.';
END
GO

-- 2. CREAR LA BASE DE DATOS
CREATE DATABASE RedSocialDB;
GO

-- 3. USAR LA NUEVA BASE DE DATOS
USE RedSocialDB;
GO

/*
================================================================
CREACIÓN DE TABLAS (100% Normalizadas)
================================================================
*/

PRINT 'Creando tablas...';

CREATE TABLE Usuarios (
    UsuarioID INT PRIMARY KEY IDENTITY(1,1),
    NombreUsuario VARCHAR(50) NOT NULL,
    Email VARCHAR(100) NOT NULL,
    FechaRegistro DATETIME DEFAULT GETDATE(),
    CONSTRAINT UQ_Usuarios_NombreUsuario UNIQUE(NombreUsuario),
    CONSTRAINT UQ_Usuarios_Email UNIQUE(Email)
);

CREATE TABLE Publicaciones (
    PublicacionID INT PRIMARY KEY IDENTITY(1,1),
    UsuarioID INT NOT NULL,
    TextoContenido VARCHAR(1000) NOT NULL,
    FechaPublicacion DATETIME DEFAULT GETDATE(),
    CONSTRAINT FK_Publicaciones_Usuarios FOREIGN KEY (UsuarioID)
        REFERENCES Usuarios(UsuarioID)
        ON DELETE CASCADE
);

CREATE TABLE Comentarios (
    ComentarioID INT PRIMARY KEY IDENTITY(1,1),
    PublicacionID INT NOT NULL,
    UsuarioID INT NOT NULL,
    TextoComentario VARCHAR(500) NOT NULL,
    FechaComentario DATETIME DEFAULT GETDATE(),
    CONSTRAINT FK_Comentarios_Publicaciones FOREIGN KEY (PublicacionID)
        REFERENCES Publicaciones(PublicacionID)
        ON DELETE CASCADE,
    CONSTRAINT FK_Comentarios_Usuarios FOREIGN KEY (UsuarioID)
        REFERENCES Usuarios(UsuarioID)
        ON DELETE NO ACTION
);

CREATE TABLE Likes (
    PublicacionID INT NOT NULL,
    UsuarioID INT NOT NULL,
    FechaLike DATETIME DEFAULT GETDATE(),
    CONSTRAINT PK_Likes PRIMARY KEY (PublicacionID, UsuarioID),
    CONSTRAINT FK_Likes_Publicaciones FOREIGN KEY (PublicacionID)
        REFERENCES Publicaciones(PublicacionID)
        ON DELETE CASCADE,
    CONSTRAINT FK_Likes_Usuarios FOREIGN KEY (UsuarioID)
        REFERENCES Usuarios(UsuarioID)
        ON DELETE NO ACTION
);

CREATE TABLE Seguidores (
    SeguidorID INT NOT NULL,
    SeguidoID INT NOT NULL,
    FechaSeguimiento DATETIME DEFAULT GETDATE(),
    CONSTRAINT PK_Seguidores PRIMARY KEY (SeguidorID, SeguidoID),
    CONSTRAINT FK_Seguidores_Seguidor FOREIGN KEY (SeguidorID)
        REFERENCES Usuarios(UsuarioID)
        ON DELETE CASCADE,
    CONSTRAINT FK_Seguidores_Seguido FOREIGN KEY (SeguidoID)
        REFERENCES Usuarios(UsuarioID)
        ON DELETE NO ACTION,
    CONSTRAINT CK_Seguidores_NoSeguirseASiMismo CHECK (SeguidorID != SeguidoID)
);

CREATE TABLE AuditoriaUsuarios (
    AuditoriaID INT PRIMARY KEY IDENTITY(1,1),
    UsuarioID INT NULL,
    CampoModificado VARCHAR(50) NOT NULL,
    ValorAntiguo VARCHAR(255),
    ValorNuevo VARCHAR(255),
    FechaModificacion DATETIME DEFAULT GETDATE(),
    CONSTRAINT FK_Auditoria_Usuarios FOREIGN KEY (UsuarioID)
        REFERENCES Usuarios(UsuarioID)
        ON DELETE SET NULL
);
PRINT 'Tablas creadas exitosamente.';
GO


```
